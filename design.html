// Electron 主进程 - 创建透明窗口 const createTodoWindow = () => { todoWindow = new BrowserWindow({ width: 320, height: 480, frame: false, transparent: true, alwaysOnTop: true, webPreferences: { nodeIntegration: true, contextIsolation: false } }); todoWindow.loadFile('index.html'); todoWindow.setOpacity(0.85); todoWindow.setAlwaysOnTop(true, 'screen-saver'); }; // Vue 组件 - 透明度控制 export default { data() { return { opacity: 0.85, todos: [] } }, methods: { updateOpacity(value) { this.opacity = value / 100; ipcRenderer.send('set-opacity', this.opacity); }, addTodo(text) { if (text.trim()) { this.todos.push({ id: Date.now(), text: text.trim(), completed: false }); } } } }
